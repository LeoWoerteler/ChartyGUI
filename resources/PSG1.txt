# Grammar:
S  -> NP VP

NP -> N
NP -> Art N
NP -> Art Adj N
NP -> Adj N
NP -> Art N PP
NP -> Art Adj N PP
NP -> Art N NP

Adj -> Adj Adj

VP -> V NP
VP -> V
VP -> V PP
VP -> V NP PP
VP -> Adv V NP

PP -> P NP
